# FSMP-kMC parameters: TMA monolayer, chicken-wire (honeycomb) structure,
# simplified 2020 model. Run from the repository root:
#   ./fsmp.out configs/tma_acid_cw.txt
# Paths are relative to the working directory; output is written there too.
#
# The chicken-wire polymorph is metastable in this model (the stable phase is
# HCP), so the stabilization mask below is essential: without it gas molecules
# fill the hexagon pores and denser phases nucleate at the interface, and the
# structure decays.

# Crystal temperature loop, K
temp_from = 300
temp_to = 301
temp_step = 10

# External field strength loop, J/mol (single point at u_m = 0)
um_from = -1000.0
um_to = -21000.0
um_step = 2000.0

# Temperature in the transition zone, K
temperature_in_transition_zone = 900
lambdam = 0.0

# Length of the MC run: total Monte Carlo steps and steps for relaxation
nSteps = 500000
nStepsEq = 250000

constant_pressure = false
constant_pressure_value = 0
kMC = true

# Keep the metastable chicken-wire framework intact: a periodic lattice of
# free wells built from the initial structure. Off-site positions (pore
# centers, HCP arrangements at the interface) cost up to mask_penalty; the
# penalty is damped by lambda(x) and vanishes in the ideal gas phase.
stabilization_mask = true
# Optional tuning (defaults shown): no penalty within mask_free_radius (A) of
# a lattice site, smooth rise over mask_ramp_width (A) up to mask_penalty (J/mol)
# mask_free_radius = 3.0
# mask_ramp_width = 2.0
# mask_penalty = 25000.0

# Numerical potential (binary v2; see forcefields/readme.txt)
potential = forcefields/TMA_simple_2020.v2.bin

# Initial structure: chicken-wire cell optimized for the simplified 2020 model
structure = TMA_CW_simple_2020

# Reference area for the chemical potential (sigma in A)
sigma_mode = manual
sigma = 11.052

# Unit cells along x and y (4 molecules per cell -> 440 molecules), and the
# fraction of free (gas) space in the cell
uc_in_x = 22
uc_in_y = 5
free_space = 0.24

# Atomistic molecule model (xyz) used for all visualizations
molecule_model = models/trimesic_acid.xyz

# MC move sizes: maximal shift (A) and rotation (deg)
delta = 2.0
delta_angle = 60.0

widom_test_index = false

# Output files (xyz animation of the cell optimization, xyz trajectory)
unit_cell_name = 0_calculate_animation.xyz
xyz_name = 1_xyz_for_calculations_TMA_cw.xyz
statistics_name = 2_my_statistics_cw.dat
